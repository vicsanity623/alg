import os
import threading

# Command to start the server
# Note: Since the model is on your local device, this script will 
# wait for you to upload it to Colab's file explorer.
model_path = "/content/model.gguf" 

def run_it():
    os.system(f"python3 -m llama_cpp.server --model {model_path} --n_gpu_layers -1 --host 0.0.0.0 --port 8000")

threading.Thread(target=run_it).start()
